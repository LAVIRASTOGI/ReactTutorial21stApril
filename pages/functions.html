Functions in JavaScript

Functions are one of the fundamental building blocks in JavaScript. A function is a reusable block of code designed to perform a particular task.

JavaScript offers several ways to create and work with functions, including traditional function declarations, function expressions, and arrow functions (ES6).

Function Declaration

The most common way to define a function:

// Function declaration
function greet(name) {
  return `Hello, ${name}!`;
}

// Function call
console.log(greet('Alice')); // "Hello, Alice!"

// Function declarations are hoisted
console.log(subtract(10, 5)); // 5 (works before declaration)

function subtract(a, b) {
  return a - b;
}

Function Expression

Assigning a function to a variable:

// Function expression (anonymous function)
const add = function(a, b) {
  return a + b;
};

console.log(add(5, 3)); // 8

// Named function expression
const multiply = function multiplyFunc(a, b) {
  return a * b;
};

console.log(multiply(4, 3)); // 12

// Function expressions are not hoisted
// console.log(divide(10, 2)); // Error: divide is not a function

const divide = function(a, b) {
  return a / b;
};

Arrow Functions (ES6)

A shorter syntax introduced in ES6:

// Basic arrow function
const square = (num) => {
  return num * num;
};

console.log(square(5)); // 25

// Simplified arrow function (implicit return)
const double = num => num * 2;

console.log(double(4)); // 8

// Arrow function with multiple parameters
const sum = (a, b, c) => a + b + c;

console.log(sum(1, 2, 3)); // 6

// Arrow function with no parameters
const getRandomNumber = () => Math.random();

console.log(getRandomNumber()); // Random number between 0 and 1

Differences Between Regular and Arrow Functions

Arrow functions differ from traditional functions in several important ways: 

// 1. 'this' binding
// Regular function: 'this' is dynamic and depends on how the function is called
const person = {
  name: 'John',
  regularFunc: function() {
    console.log(this.name); // 'John' - 'this' refers to the person object
  },
  arrowFunc: () => {
    console.log(this.name); // undefined - 'this' is inherited from the surrounding scope
  }
};

// 2. No 'arguments' object in arrow functions
function regularFunc() {
  console.log(arguments); // [Arguments] { '0': 1, '1': 2, '2': 3 }
}

const arrowFunc = () => {
  // console.log(arguments); // Error: arguments is not defined
};

// 3. Cannot be used as constructors
function Person(name) {
  this.name = name;
}
const john = new Person('John');

// const ArrowPerson = (name) => { this.name = name; };
// const alice = new ArrowPerson('Alice'); // Error: ArrowPerson is not a constructor

// 4. No prototype property
console.log(Person.prototype); // {} (exists)
// console.log(arrowFunc.prototype); // undefined

Function Parameters

JavaScript functions can handle parameters in various ways:

// Default parameters (ES6)
function greetWithDefault(name = 'Guest') {
  return `Hello, ${name}!`;
}

console.log(greetWithDefault()); // "Hello, Guest!"
console.log(greetWithDefault('Jane')); // "Hello, Jane!"

// Rest parameters (ES6)
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3, 4, 5)); // 15

// Parameter destructuring (ES6)
function printPersonInfo({ name, age, city = 'Unknown' }) {
  console.log(`${name} is ${age} years old and lives in ${city}.`);
}

printPersonInfo({ name: 'Alice', age: 30 }); // "Alice is 30 years old and lives in Unknown."
printPersonInfo({ name: 'Bob', age: 25, city: 'New York' }); // "Bob is 25 years old and lives in New York."

Immediately Invoked Function Expressions (IIFE)

Functions that run as soon as they are defined:

// IIFE with function declaration
(function() {
  const secret = 'I am private';
  console.log(secret); // "I am private"
})();

// IIFE with arrow function
(() => {
  const anotherSecret = 'Also private';
  console.log(anotherSecret); // "Also private"
})();

// IIFE with parameters
((name) => {
  console.log(`Hello, ${name}!`); // "Hello, World!"
})('World');

Higher-Order Functions

Functions that accept or return other functions:

// Function that returns a function
function createMultiplier(factor) {
  return function(number) {
    return number * factor;
  };
}

const double = createMultiplier(2);
const triple = createMultiplier(3);

console.log(double(5)); // 10
console.log(triple(5)); // 15

// Function that accepts a function as an argument
function applyOperation(a, b, operation) {
  return operation(a, b);
}

const add = (x, y) => x + y;
const subtract = (x, y) => x - y;

console.log(applyOperation(10, 5, add)); // 15
console.log(applyOperation(10, 5, subtract)); // 5
