Spread and Rest Operators in JavaScript

The spread (...) and rest operators are powerful features introduced in ES6 (ES2015). Despite sharing the same syntax (...), they serve different purposes depending on how they're used.

They help simplify and write more concise code when working with arrays, objects, and function parameters.

Spread Operator with Arrays

The spread operator expands an array into individual elements:

// 1. Copying an array
const originalArray = [1, 2, 3];
const copyArray = [...originalArray];
console.log(copyArray); // [1, 2, 3]

// 2. Merging arrays
const array1 = [1, 2, 3];
const array2 = [4, 5, 6];
const mergedArray = [...array1, ...array2];
console.log(mergedArray); // [1, 2, 3, 4, 5, 6]

// 3. Creating a new array with additional elements
const numbers = [1, 2, 3];
const expandedNumbers = [0, ...numbers, 4, 5];
console.log(expandedNumbers); // [0, 1, 2, 3, 4, 5]

// 4. Converting string to array of characters
const str = "hello";
const chars = [...str];
console.log(chars); // ['h', 'e', 'l', 'l', 'o']

// 5. Using with Math functions
const nums = [5, 3, 8, 1, 4];
console.log(Math.max(...nums)); // 8 (equivalent to Math.max(5, 3, 8, 1, 4))

Spread Operator with Objects

The spread operator works with objects to create shallow copies and merge objects: 

// 1. Copying an object (shallow copy)
const originalObj = { name: 'Alice', age: 30 };
const copyObj = { ...originalObj };
console.log(copyObj); // { name: 'Alice', age: 30 }

// 2. Merging objects
const person = { name: 'Bob', age: 25 };
const details = { job: 'Developer', city: 'New York' };
const mergedPerson = { ...person, ...details };
console.log(mergedPerson); 
// { name: 'Bob', age: 25, job: 'Developer', city: 'New York' }

// 3. Creating a new object with modifications
const user = { id: 1, name: 'Charlie', email: 'charlie@example.com' };
const updatedUser = { ...user, email: 'charlie.new@example.com', role: 'Admin' };
console.log(updatedUser);
// { id: 1, name: 'Charlie', email: 'charlie.new@example.com', role: 'Admin' }

// 4. Property order matters for duplicated properties
const obj1 = { a: 1, b: 2 };
const obj2 = { b: 3, c: 4 }; // Note that 'b' is duplicated
const merged1 = { ...obj1, ...obj2 }; // obj2 properties override obj1
console.log(merged1); // { a: 1, b: 3, c: 4 }

const merged2 = { ...obj2, ...obj1 }; // obj1 properties override obj2
console.log(merged2); // { b: 2, c: 4, a: 1 }

Rest Operator with Arrays

The rest operator collects multiple elements into an array. It's often used in destructuring: 

// 1. Collecting remaining array elements
const [first, second, ...remaining] = [1, 2, 3, 4, 5];
console.log(first);     // 1
console.log(second);    // 2
console.log(remaining); // [3, 4, 5]

// 2. Skipping elements
const [head, ...tail] = [10, 20, 30, 40, 50];
console.log(head); // 10
console.log(tail); // [20, 30, 40, 50]

// 3. Collecting specific parts
const scores = [85, 92, 78, 90, 88];
const [highest, ...otherScores] = scores.sort((a, b) => b - a);
console.log(highest);     // 92
console.log(otherScores); // [90, 88, 85, 78]

Rest Operator with Objects

The rest operator can collect remaining object properties during destructuring: 

// 1. Collecting remaining object properties
const person = {
  name: 'Alice',
  age: 30,
  job: 'Engineer',
  city: 'San Francisco',
  country: 'USA'
};

const { name, age, ...otherDetails } = person;
console.log(name);          // 'Alice'
console.log(age);           // 30
console.log(otherDetails);  // { job: 'Engineer', city: 'San Francisco', country: 'USA' }

// 2. Extracting specific properties
const product = {
  id: 'P123',
  name: 'Laptop',
  price: 999,
  brand: 'TechBrand',
  specs: {
    cpu: 'i7',
    ram: '16GB',
    storage: '512GB'
  }
};

const { id, name: productName, ...productDetails } = product;
console.log(id);             // 'P123'
console.log(productName);    // 'Laptop'
console.log(productDetails); // { price: 999, brand: 'TechBrand', specs: {...} }

Rest Parameters in Functions

The rest parameter syntax allows functions to accept an indefinite number of arguments as an array: 

// 1. Collecting all arguments
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3));       // 6
console.log(sum(5, 10, 15, 20)); // 50

// 2. Collecting remaining arguments
function calculate(operation, ...numbers) {
  if (operation === 'sum') {
    return numbers.reduce((total, num) => total + num, 0);
  } else if (operation === 'multiply') {
    return numbers.reduce((result, num) => result * num, 1);
  }
  return null;
}

console.log(calculate('sum', 1, 2, 3, 4));        // 10
console.log(calculate('multiply', 2, 3, 4));      // 24

// 3. Combining with regular parameters
function formatName(title, firstName, lastName, ...titles) {
  return `${title} ${firstName} ${lastName}${titles.length ? ', ' + titles.join(', ') : ''}`;
}

console.log(formatName('Mr.', 'John', 'Doe')); // "Mr. John Doe"
console.log(formatName('Dr.', 'Jane', 'Smith', 'PhD', 'Professor')); // "Dr. Jane Smith, PhD, Professor"

Practical Use Cases

Common real-world applications of spread and rest operators:

// 1. Cloning and updating immutable objects (common in React/Redux)
function updateUser(user, updates) {
  return { ...user, ...updates };
}

const user = { id: 1, name: 'Alice', email: 'alice@example.com' };
const updatedUser = updateUser(user, { email: 'newalice@example.com' });
console.log(updatedUser); // { id: 1, name: 'Alice', email: 'newalice@example.com' }

// 2. Adding items to an array without mutating the original
function addItem(array, item) {
  return [...array, item];
}

const fruits = ['apple', 'banana', 'orange'];
const newFruits = addItem(fruits, 'grape');
console.log(fruits);     // ['apple', 'banana', 'orange'] (unchanged)
console.log(newFruits);  // ['apple', 'banana', 'orange', 'grape']

// 3. Function composition with multiple arguments
function applyFunctions(value, ...functions) {
  return functions.reduce((result, func) => func(result), value);
}

const double = x => x * 2;
const increment = x => x + 1;
const square = x => x * x;

console.log(applyFunctions(3, double, increment, square)); // 49 ((3*2+1)²)

// 4. Converting NodeList to Array
// document.querySelectorAll() returns a NodeList, not an Array
// const elements = document.querySelectorAll('.some-class');
// const elementsArray = [...elements]; // Now it's a real array with all array methods

// 5. Removing specific properties from an object (via destructuring and rest)
function removeProperty(obj, prop) {
  const { [prop]: _, ...rest } = obj;
  return rest;
}

const config = { server: 'localhost', port: 8080, timeout: 1000 };
const configWithoutTimeout = removeProperty(config, 'timeout');
console.log(configWithoutTimeout); // { server: 'localhost', port: 8080 }
